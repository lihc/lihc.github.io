Git 命令总结
白羊
 2021-09-05 14:06:44  
  • Other 
 
  • Git 
 

https://training.github.com/downloads/zh_CN/github-git-cheat-sheet/

#初始化仓库
git init

#提交文件到暂存区
git add <file>

#放弃提交文件并撤销修改
git restore <file>

#暂存区文件上传到仓库
git commit -m "<说明>"

#查看仓库状态
git status

#查看版本日志
git log
#查看操作日志
git reflog

#回退版本
git reset --hard HEAD^ #回退上一个版本
git reset --hard <commit> #回退指定的版本


#查看分支
git branch
#创建分支
git branch <分支>
#删除分支
git branch -d <分支>

#切换分支
git switch <分支>
git checkout <分支>
#合并分支
git merge <分支>


#添加远程仓库
git remote add <origin> <仓库地址>

#修改远程仓库
git remote set-url origin url

#查看远程仓库列表
git remote -v

#克隆远程仓库
git clone <仓库地址>
#上传文件到远程仓库
git push -u <origin> <master>
 -f 强制上传
#初始化仓库
git init

#连接远程仓库
git remote add origin [url]

#克隆仓库
git clone [url]

#配置用户名
git config --global user.name "[name]"

#配置邮箱地址
git config --global user.email "[email address]"

#创建新分支
git branch [branch_name]

#切换分支并更新工作目录
git switch -c [branch_name]

#将指定分支合并到当前分支
git merge [branch]

#删除指定分支
git branch -d [branch_name]

#查看当前分支版本历史
git log

#查看文件的版本历史
git log --follow [file]

#查看分支的内容差异
git diff [first_branch]...[second_branch]

#输出指定commit的元数据和内容变化
git show [commit]

#添加文件到暂存区
git add [file]

#提交暂存区文件到仓库
git commit -m "[descriptive message]"
