Shell 脚本的最佳实践

#!/usr/bin/env bash

1. 定义变量

PATH=${PATH:-'/usr/local/sbin:/usr/local/bin:/usr/sbin:/usr/bin'}

${var:-word}
${var:=word}
${var:?word}
${var:+word}

$0
$#
$*
$?
$$
$!
$_
DI="$(cd "$(dirname "$0")" && pwd)"

Shell 中的特殊变量：

# 定义环境变量
export PATH=/usr/local/sbin:/usr/local/bin:/usr/sbin:/usr/bin:/root/bin

# 定义超时时间
export TMOUT=300

# 定义 Bash 提示符
export PS1='[\u@\h \W]\$'

# 定义历史记录格式
export HISTTIMEFORMAT='%F %T '
history
1  2020-01-01 12:00:00 exit

# 定义提示符(PS1)出现之前的命令
export PROMPT_COMMAND='echo hello'

# 获取当前函数名
main () { echo ${FUNCNAME}; }
main

# 获取当前执行行号
echo $LINENO

# 获取当前工作目录
echo $PWD

# 获取脚本运行的时间
echo $SECONDS

# 获取 Shell 的版本
for((i=0;i<=5;i++));do echo "BASH_VERSINFO[$i]=${BASH_VERSINFO[$i]}"; done

# 捕获正则匹配数组
[[ "Docker version 20.10.11" =~ ([0-9]+)\.([0-9]+)\.([0-9]+) ]]
for((i=0;i<=3;i++));do echo "BASH_REMATCH[$i]=${BASH_REMATCH[$i]}"; done

# 获取随机数(0-32767)
echo $RANDOM

2. 检查是否以 root 身份运行

check_if_running_as_root() {
  if [[ "$UID" -ne '0' ]]; then
    echo "WARNING: The user currently executing this script is not root."
    read -r -p "Are you sure you want to continue? [y/n] " enter_information
    if [[ "${enter_information:0:1}" = 'y' ]]; then
      echo "Continuing the installation with current user..."
    else
      echo "Not running with root, exiting..."
      exit 1
    fi
  fi
}
check_if_running_as_root

3. 输出文本高亮显示

#!/bin/bash
RESET=$(tput sgr0)          # 重置所有属性('\033[00m')
BLACK=$(tput setaf 0)       # 黑色,('\033[30m')
RED=$(tput setaf 1)         # 红色,('\033[31m')
GREEN=$(tput setaf 2)       # 绿色,('\033[32m')
YELLOW=$(tput setaf 3)      # 黄色,('\033[33m')
BLUE=$(tput setaf 4)        # 蓝色,('\033[34m')
MAGENTA=$(tput setaf 5)     # 品红,('\033[35m')
CYAN=$(tput setaf 6)        # 青色,('\033[36m')
WHITE=$(tput setaf 7)       # 白色,('\033[37m')
BOLD=$(tput bold)           # 加粗,('\033[01m')
BLINK=$(tput blink)         # 闪烁,('\033[05m')
SMSO=$(tput smso)           # 高亮,('\033[07m')
SMUL=$(tput smul)           # 下划线,('\033[04m')
CIVIS=$(tput civis)         # 隐藏光标,('\033[?25l')
CNORM=$(tput c)         # 显示光标,('\033[?25h')

log::error () {
    printf "${RED}[ERROR]${RESET} %b\n" "$@"
}
log::warning () {
    printf "${YELLOW}[WARNGIN]${RESET} %b\n" "$@"
}
log::info () {
    printf "${GREEN}[INFO]${RESET} %b\n" "$@"
}
log::debug () {
    printf "${WHITE}[DEBUG]${RESET} %b\n" "$@"
}

4. 打印菜单

menu() {
cat <<EOF
1) 查看当前用户
2) 查看主机名称
3) 查看环境变量
EOF
read -p "输入序号:" num
case $num in
  1)
  echo $USER
  ;;
  2)
  echo $HOSTNAME
  ;;
  3)
  echo $PATH
  ;;
  *)
  echo '*'
esac
}
menu

5. 数学运算

expr 1 + 5

var1=$[ 1 + 5 ]

6. 读取文件每一行内容

read_line ()
{
    while read -r line ; do
        echo "${line}"
    done < "${FILE}"
}
FILE=""
read_line

7. 分支结构 (case in)

case '' in
  '-a | -all')
  print aaa
  ;;
  *)
  echo ""
  ;;
esac

8. 捕获中断 (trap)

control_c () {
    printf "\nCtrl-C,GoodBye!\n"
    exit 1
}
trap control_c SIGINT
